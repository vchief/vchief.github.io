jQuery Mobile 触摸事件

jQuery Mobile 点击
$("p").on("tap",function(){
  $(this).hide();
});

jQuery Mobile 点击不放（长按）
$("p").on("taphold",function(){
  $(this).hide();
});

jQuery Mobile 滑动
$("p").on("swipe",function(){
  $("span").text("滑动检测!");
});

jQuery Mobile 向左滑动
$("p").on("swipeleft",function(){
  alert("向左滑动!");
});

jQuery Mobile 向右滑动
$("p").on("swiperight",function(){
  alert("向右滑动!");
});
